// BCNF RULES

//parser starts here
Statements ::= ( Statement )*
Statement ::= Assignment
| FunctionDeclaration
| FunctionCall
| IfStatment
| whileLoop
| returnStatement
| SimpleStatement
| forLoop
Assignment ::= ( DataType )? <IDENTIFIER> ( <EQUAL> ExpressionOrSingleType )? <SEMICOLON>
FunctionDeclaration ::= DataType <IDENTIFIER> <LEFT_PAREN> ( DataType ExpressionOrSingleType ( <COMMA> )? )* <RIGHT_PAREN> <LEFT_BRACKET> Statements <RIGHT_BRACKET>
FunctionCall ::= <IDENTIFIER> <LEFT_PAREN> ( ExpressionOrSingleType ( <COMMA> )? )* <RIGHT_PAREN> <SEMICOLON> 
//statments such as (z++;, z*z;)
SimpleStatement ::= ExpressionOrSingleType <SEMICOLON>
IfStatment ::= If_Part ( Else_Part )?
If_Part ::= <IF> <LEFT_PAREN> Conditions <RIGHT_PAREN> <LEFT_BRACKET> Statements <RIGHT_BRACKET>
Else_Part ::= <ELSE> <LEFT_BRACKET> Statements <RIGHT_BRACKET>
whileLoop ::= <WHILE> <LEFT_PAREN> Conditions <RIGHT_PAREN> <LEFT_BRACKET> Statements <RIGHT_BRACKET>
forLoop ::= <FOR> <LEFT_PAREN> Assignment Conditions <SEMICOLON> forInc <RIGHT_PAREN> <LEFT_BRACKET> Statements <RIGHT_BRACKET> 
//need to add support for statments like i = i+1
forInc ::= <IDENTIFIER> ( incOrDec )
returnStatement ::= <RETURN> ExpressionOrSingleType <SEMICOLON> 
//CONDITIONS eg ( a == 9, b <4  ect
Conditions ::= ExpressionOrSingleType booleanOperators ExpressionOrSingleType
DataType ::= <VOID>
| <INT>
| <DOUBLE>
| <CHAR>
booleanOperators ::= <EQUAL_BOOL>
| <GT>
| <LT>
| <LTE>
| <GTE>
ExpressionOrSingleType ::= Expression
| assignmentType 
//types that can be assigned to a variable
assignmentType ::= ( <INTEGER> | <REAL> | <IDENTIFIER> | <CHARACTER> ) ( incOrDec )*
| <STRING>
Expression ::= assignmentType ( arethmeticOperators assignmentType )*
arethmeticOperators ::= <PLUS>
| <MINUS>
| <MULTIPLY>
| <DIVIDE>
incOrDec ::= <INC>
| <DEC>
identifier ::= <IDENTIFIER>
